00035 // class TypeInfo
00036 // Purpose: offer a first-class, comparable wrapper over std::type_info
00038 
00039     class TypeInfo
00040     {
00041     public:
00042         // Constructors
00043         TypeInfo(); // needed for containers
00044         TypeInfo(const std::type_info&); // non-explicit
00045 
00046         // Access for the wrapped std::type_info
00047         const std::type_info& Get() const;
00048         // Compatibility functions
00049         bool before(const TypeInfo& rhs) const;
00050         const char* name() const;
00051 
00052     private:
00053         const std::type_info* pInfo_;
00054     };
00055     
00056 // Implementation
00057     
00058     inline TypeInfo::TypeInfo()
00059     {
00060         class Nil {};
00061         pInfo_ = &typeid(Nil);
00062         assert(pInfo_);
00063     }
00064     
00065     inline TypeInfo::TypeInfo(const std::type_info& ti)
00066     : pInfo_(&ti)
00067     { assert(pInfo_); }
00068     
00069     inline bool TypeInfo::before(const TypeInfo& rhs) const
00070     {
00071         assert(pInfo_);
00072         return pInfo_->before(*rhs.pInfo_);
00073     }
00074 
00075     inline const std::type_info& TypeInfo::Get() const
00076     {
00077         assert(pInfo_);
00078         return *pInfo_;
00079     }
00080     
00081     inline const char* TypeInfo::name() const
00082     {
00083         assert(pInfo_);
00084         return pInfo_->name();
00085     }
00086 
00087 // Comparison operators
00088     
00089     inline bool operator==(const TypeInfo& lhs, const TypeInfo& rhs)
00090     { return lhs.Get() == rhs.Get(); }
00091 
00092     inline bool operator<(const TypeInfo& lhs, const TypeInfo& rhs)
00093     { return lhs.before(rhs); }
00094 
00095     inline bool operator!=(const TypeInfo& lhs, const TypeInfo& rhs)
00096     { return !(lhs == rhs); }    
00097     
00098     inline bool operator>(const TypeInfo& lhs, const TypeInfo& rhs)
00099     { return rhs < lhs; }
00100     
00101     inline bool operator<=(const TypeInfo& lhs, const TypeInfo& rhs)
00102     { return !(lhs > rhs); }
00103
