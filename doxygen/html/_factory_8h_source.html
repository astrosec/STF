00036 // class template DefaultFactoryError
00037 // Manages the "Unknown Type" error in an object factory
00039 
00040     template <typename IdentifierType, class AbstractProduct>
00041     struct DefaultFactoryError
00042     {
00043         struct Exception : public std::exception
00044         {
00045             const char* what() const { return "Unknown Type"; }
00046         };
00047         
00048         static AbstractProduct* OnUnknownType(IdentifierType)
00049         {
00050             throw Exception();
00051         }
00052     };

00055 // class template Factory
00056 // Implements a generic object factory
00058 
00059     template
00060     <
00061         class AbstractProduct, 
00062         typename IdentifierType,
00063         typename ProductCreator = AbstractProduct* (*)(),
00064         template<typename, class>
00065             class FactoryErrorPolicy = DefaultFactoryError
00066     >
00067     class Factory 
00068         : public FactoryErrorPolicy<IdentifierType, AbstractProduct>
00069     {
00070     public:
00071         bool Register(const IdentifierType& id, ProductCreator creator)
00072         {
00073             return associations_.insert(
00074                 IdToProductMap::value_type(id, creator)).second;
00075         }
00076         
00077         bool Unregister(const IdentifierType& id)
00078         {
00079             return associations_.erase(id) == 1;
00080         }
00081         
00082         AbstractProduct* CreateObject(const IdentifierType& id)
00083         {
00084             typename IdToProductMap::const_iterator i = associations_.find(id);
00085             if (i != associations_.end())
00086             {
00087                 return (i->second)();
00088             }
00089             return OnUnknownType(id);
00090         }
00091         
00092     private:
00093         typedef AssocVector<IdentifierType, ProductCreator> IdToProductMap;
00094         IdToProductMap associations_;
00095     };
