00218 {
00219     if (allocChunk_ == 0 || allocChunk_->blocksAvailable_ == 0)
00220     {
00221         Chunks::iterator i = chunks_.begin();
00222         for (;; ++i)
00223         {

00232                 deallocChunk_ = &chunks_.front();
00233                 break;

00271 FixedAllocator::Chunk* FixedAllocator::VicinityFind(void* p)
00272 {
00273     assert(!chunks_.empty());
00274     assert(deallocChunk_);
00275 
00276     const std::size_t chunkLength = numBlocks_ * blockSize_;
00277 
00278     Chunk* lo = deallocChunk_;
00279     Chunk* hi = deallocChunk_ + 1;
00280     Chunk* loBound = &chunks_.front();
00281     Chunk* hiBound = &chunks_.back() + 1;
00282 
00283     for (;;)
00284     {
00285         if (lo)
00286         {
00287             if (p >= lo->pData_ && p < lo->pData_ + chunkLength)
00288             {
00289                 return lo;
00290             }
00291             if (lo == loBound) lo = 0;
00292             else --lo;
00293         }
00294         
00295         if (hi)

00297             if (p >= hi->pData_ && p < hi->pData_ + chunkLength)
00298             {
00299                 return hi;

00309 // FixedAllocator::DoDeallocate (internal)
00310 // Performs deallocation. Assumes deallocChunk_ points to the correct chunk
00312 
00313 void FixedAllocator::DoDeallocate(void* p)
00314 {
00315     assert(deallocChunk_->pData_ <= p);
00316     assert(deallocChunk_->pData_ + numBlocks_ * blockSize_ > p);
00317 
00318     // call into the chunk, will adjust the inner list but won't release memory
00319     deallocChunk_->Deallocate(p, blockSize_);
00320 
00321     if (deallocChunk_->blocksAvailable_ == numBlocks_)
00322     {
00323         // deallocChunk_ is completely free, should we release it? 
00324         
00325         Chunk& lastChunk = chunks_.back();
00326         
00327         if (&lastChunk == deallocChunk_)
00328         {
00329             // check if we have two last chunks empty
00330             
00331             if (chunks_.size() > 1 && 
00332                 deallocChunk_[-1].blocksAvailable_ == numBlocks_)
00333             {
00334                 // Two free chunks, discard the last one
00335                 lastChunk.Release();
00336                 chunks_.pop_back();
00337                 allocChunk_ = deallocChunk_ = &chunks_.front();
00338             }
00339             return;
00340         }
00341         
00342         if (lastChunk.blocksAvailable_ == numBlocks_)
00343         {
00344             // Two free blocks, discard one
00345             lastChunk.Release();
00346             chunks_.pop_back();
00347             allocChunk_ = deallocChunk_;
00348         }
00349         else
00350         {
00351             // move the empty chunk to the end
00352             std::swap(*deallocChunk_, lastChunk);
00353             allocChunk_ = &chunks_.back();
00354         }
00355     }
00356 }

00374 // Allocates 'numBytes' memory
00375 // Uses an internal pool of FixedAllocator objects for small objects  
00377 
00378 void* SmallObjAllocator::Allocate(std::size_t numBytes)
00379 {
00380     if (numBytes > maxObjectSize_) return operator new(numBytes);
00381     
00382     if (pLastAlloc_ && pLastAlloc_->BlockSize() == numBytes)
00383     {
00384         return pLastAlloc_->Allocate();
00385     }
00386     Pool::iterator i = std::lower_bound(pool_.begin(), pool_.end(), numBytes);
00387     if (i == pool_.end() || i->BlockSize() != numBytes)

00389         i = pool_.insert(i, FixedAllocator(numBytes));
