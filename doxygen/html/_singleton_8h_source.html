00102 // function template SetLongevity
00103 // Assigns an object a longevity; ensures ordered destructions of objects 
00104 //     registered thusly during the exit sequence of the application
00106 
00107     template <typename T, typename Destroyer>
00108     void SetLongevity(T* pDynObject, unsigned int longevity,
00109         Destroyer d = Private::Deleter<T>::Delete)
00110     {
00111         using namespace Private;
00112         
00113         TrackerArray pNewArray = static_cast<TrackerArray>(
00114                 std::realloc(pTrackerArray, elements + 1));
00115         if (!pNewArray) throw std::bad_alloc();
00116         
00117         LifetimeTracker* p = new ConcreteLifetimeTracker<T, Destroyer>(
00118             pDynObject, longevity, d);
00119         
00120         // Delayed assignment for exception safety
00121         pTrackerArray = pNewArray;
00122         
00123         // Insert a pointer to the object into the queue
00124         TrackerArray pos = std::upper_bound(
00125             pTrackerArray, 
00126             pTrackerArray + elements, 
00127             p, 
00128             LifetimeTracker::Compare);
00129         std::copy_backward(
00130             pos, 
00131             pTrackerArray + elements,
00132             pTrackerArray + elements + 1);
00133         *pos = p;
00134         ++elements;
00135         
00136         // Register a call to AtExitFn
00137         std::atexit(Private::AtExitFn);
00138     }

00402     template

00404         class T,
00405         template <class> class CreationPolicy,
00406         template <class> class LifetimePolicy,
00407         template <class> class ThreadingModel

00409     void SingletonHolder<T, CreationPolicy, 
00410         LifetimePolicy, ThreadingModel>::MakeInstance()
00411     {
00412         typename ThreadingModel<T>::Lock guard;
00413         (void)guard;
00414         
00415         if (!pInstance_)
00416         {
00417             if (destroyed_)
00418             {
00419                 LifetimePolicy<T>::OnDeadReference();
00420                 destroyed_ = false;
00421             }
00422             pInstance_ = CreationPolicy<T>::Create();
00423             LifetimePolicy<T>::ScheduleDestruction(pInstance_, 
00424                 &DestroySingleton);
