00094 // class template PrototypeFactoryUnit
00095 // Creates an object by cloning a prototype
00096 // There is a difference between the implementation herein and the one described
00097 //     in the book: GetPrototype and SetPrototype use the helper friend 
00098 //     functions DoGetPrototype and DoSetPrototype. The friend functions avoid
00099 //     name hiding issues. Plus, GetPrototype takes a reference to pointer
00100 //     instead of returning the pointer by value.
00102 
00103     template <class ConcreteProduct, class Base>
00104     class PrototypeFactoryUnit : public Base
00105     {
00106         typedef typename Base::ProductList BaseProductList;
00107     
00108     protected:
00109         typedef typename BaseProductList::Tail ProductList;
00110 
00111     public:
00112         typedef typename BaseProductList::Head AbstractProduct;
00113 
00114         PrototypeFactoryUnit(AbstractProduct* p = 0)
00115             : pPrototype_(p)
00116         {}
00117         
00118         friend void DoGetPrototype(const PrototypeFactoryUnit& me,
00119             AbstractProduct*& pPrototype)
00120         { pPrototype = me.pPrototype_; }
00121         
00122         friend void DoSetPrototype(PrototypeFactoryUnit& me, 
00123             AbstractProduct* pObj)
00124         { me.pPrototype_ = pObj; }
00125         
00126         template <class U>
00127         void GetPrototype(AbstractProduct*& p)
00128         { return DoGetPrototype(*this, p); }
00129         
00130         template <class U>
00131         void SetPrototype(U* pObj)
00132         { DoSetPrototype(*this, pObj); }
00133         
00134         AbstractProduct* DoCreate(Type2Type<AbstractProduct>)
00135         {
00136             assert(pPrototype_);
00137             return pPrototype_->Clone();
00138         }
00139         
00140     private:
00141         AbstractProduct* pPrototype_;
00142     };
