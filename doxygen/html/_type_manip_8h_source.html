00095 // class template Conversion
00096 // Figures out the conversion relationships between two types
00097 // Invocations (T and U are types):
00098 // a) Conversion<T, U>::exists
00099 // returns (at compile time) true if there is an implicit conversion from T
00100 // to U (example: Derived to Base)
00101 // b) Conversion<T, U>::exists2Way
00102 // returns (at compile time) true if there are both conversions from T
00103 // to U and from U to T (example: int to char and back)
00104 // c) Conversion<T, U>::sameType
00105 // returns (at compile time) true if T and U represent the same type
00106 //
00107 // Caveat: might not work if T and U are in a private inheritance hierarchy.
00109 
00110     template <class T, class U>
00111     struct Conversion
00112     {
00113         typedef Private::ConversionHelper<T, U> H;
00114 #ifndef __MWERKS__
00115         enum { exists = sizeof(typename H::Small) == sizeof(H::Test(H::MakeT())) };
00116 #else
00117         enum { exists = false };
00118 #endif
00119         enum { exists2Way = exists && Conversion<U, T>::exists };
00120         enum { sameType = false };
00121     };
00122     
00123     template <class T>
00124     struct Conversion<T, T>    
00125     {
00126         enum { exists = 1, exists2Way = 1,sameType = 1 };
00127     };
00128     
00129     template <class T>
00130     struct Conversion<void, T>    
00131     {
00132         enum { exists = 1, exists2Way = 0,sameType = 0 };
00133     };
00134     
00135     template <class T>
00136     struct Conversion<T, void>    
00137     {
00138         enum { exists = 1, exists2Way = 0,sameType = 0 };
00139     };
00140

00143     {
00144     public:
00145         enum { exists = 1, exists2Way = 1,sameType = 1 };
00146     };

00150 // macro SUPERSUBCLASS
00151 // Invocation: SUPERSUBCLASS(B, D) where B and D are types. 
00152 // Returns true if B is a public base of D, or if B and D are aliases of the 
00153 // same type.
00154 //
00155 // Caveat: might not work if T and U are in a private inheritance hierarchy.
