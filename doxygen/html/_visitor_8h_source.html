00046 // class template Visitor
00047 // The building block of Acyclic Visitor
00049 
00050     template <class T, typename R = void>
00051     class Visitor
00052     {
00053     public:
00054         typedef R ReturnType;
00055         virtual ReturnType Visit(T&) = 0;
00056     };

00059 // class template Visitor (specialization)
00060 // This specialization is not present in the book. It makes it easier to define
00061 // Visitors for multiple types in a shot by using a typelist. Example:
00062 //
00063 // class SomeVisitor : 
00064 //     public BaseVisitor // required
00065 //     public Visitor<TYPELIST_2(RasterBitmap, Paragraph)>, 
00066 //     public Visitor<Paragraph>
00067 // {
00068 // public:
00069 //     void Visit(RasterBitmap&); // visit a RasterBitmap
00070 //     void Visit(Paragraph &);   // visit a Paragraph
00071 // };
00073 
00074     template <class Head, class Tail, typename R>
00075     class Visitor<Typelist<Head, Tail>, R>
00076         : public Visitor<Head, R>, public Visitor<Tail, R>
00077     {
00078     public:
00079         typedef R ReturnType;
00080        // using Visitor<Head, R>::Visit;
00081        // using Visitor<Tail, R>::Visit;
00082     };
00083     
00084     template <class Head, typename R>
00085     class Visitor<Typelist<Head, NullType>, R> : public Visitor<Head, R>
00086     {
00087     public:
00088         typedef R ReturnType;
00089         using Visitor<Head, R>::Visit;
00090     };

00094 // Implements non-strict visitation (you can implement only part of the Visit

00122 // class template BaseVisitable
00124 
00125 template <typename R, typename Visited>
00126 struct DefaultCatchAll
00127 {
00128     static R OnUnknownVisitor(Visited&, BaseVisitor&)
00129     { return R(); }
00130 };
00131 
00133 // class template BaseVisitable
00135 
00136     template 
00137     <
00138         typename R = void, 
00139         template <typename, class> class CatchAll = DefaultCatchAll
00140     >
00141     class BaseVisitable
00142     {
00143     public:
00144         typedef R ReturnType;
00145         virtual ~BaseVisitable() {}
00146         virtual ReturnType Accept(BaseVisitor&) = 0;
00147         
00148     protected: // give access only to the hierarchy
00149         template <class T>
00150         static ReturnType AcceptImpl(T& visited, BaseVisitor& guest)
00151         {
00152             // Apply the Acyclic Visitor
00153             if (Visitor<T>* p = dynamic_cast<Visitor<T>*>(&guest))
00154             {
00155                 return p->Visit(visited);
00156             }
00157             return CatchAll<R, T>::OnUnknownVisitor(visited, guest);
00158         }
00159     };

00162 // macro DEFINE_VISITABLE
00163 // Put it in every class that you want to make visitable (in addition to 
00164 //     deriving it from BaseVisitable<R>
00166 
00167 #define DEFINE_VISITABLE() \
00168     virtual ReturnType Accept(BaseVisitor& guest) \
00169     { return AcceptImpl(*this, guest); }

00172 // class template CyclicVisitor
00173 // Put it in every class that you want to make visitable (in addition to 
00174 //     deriving it from BaseVisitable<R>
00176 
00177     template <typename R, class TList>
00178     class CyclicVisitor : public Visitor<TList, R>
00179     {
00180     public:
00181         typedef R ReturnType;
00182         // using Visitor<TList, R>::Visit;
00183         
00184         template <class Visited>
00185         ReturnType GenericVisit(Visited& host)
00186         {
00187             Visitor<Visited, ReturnType>& subObj = *this;
00188             return subObj.Visit(host);
00189         }
00190     };
