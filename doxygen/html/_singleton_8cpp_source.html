00037 void Loki::Private::AtExitFn()
00038 {
00039     assert(elements > 0 && pTrackerArray != 0);
00040     // Pick the element at the top of the stack
00041     LifetimeTracker* pTop = pTrackerArray[elements - 1];
00042     // Remove that object off the stack
00043     // Don't check errors - realloc with less memory 
00044     //     can't fail
00045     pTrackerArray = static_cast<TrackerArray>(std::realloc(
00046         pTrackerArray, --elements));
00047     // Destroy the element
00048     delete pTop;
00049 }
