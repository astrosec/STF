00072     template
00073     <
00074         class Executor,
00075         class BaseLhs, 
00076         class TypesLhs,
00077         bool symmetric = true,
00078         class BaseRhs = BaseLhs,
00079         class TypesRhs = TypesLhs,
00080         typename ResultType = void
00081     >
00082     class StaticDispatcher
00083     {
00084         template <class SomeLhs>
00085         static ResultType DispatchRhs(SomeLhs& lhs, BaseRhs& rhs,
00086             Executor exec, NullType)
00087         { return exec.OnError(lhs, rhs); }
00088         
00089         template <class TList, class SomeLhs>
00090         static ResultType DispatchRhs(SomeLhs& lhs, BaseRhs& rhs,
00091             Executor exec, TList)
00092         {
00093             typedef typename TList::Head Head;
00094             typedef typename TList::Tail Tail;
00095             
00096             if (Head* p2 = dynamic_cast<Head*>(&rhs))
00097             {
00098                 Int2Type<(symmetric &&
00099                           int(TL::IndexOf<TypesRhs, Head>::value) <
00100                           int(TL::IndexOf<TypesLhs, SomeLhs>::value))> i2t;
00101 
00102                 typedef Private::InvocationTraits< 
00103                         SomeLhs, Head, Executor, ResultType> CallTraits;
00104                     
00105                 return CallTraits::DoDispatch(lhs, *p2, exec, i2t);
00106             }
00107             return DispatchRhs(lhs, rhs, exec, Tail());
00108         }
00109         
00110         static ResultType DispatchLhs(BaseLhs& lhs, BaseRhs& rhs,
00111             Executor exec, NullType)
00112         { return exec.OnError(lhs, rhs); }
00113         
00114         template <class TList>
00115         static ResultType DispatchLhs(BaseLhs& lhs, BaseRhs& rhs,
00116             Executor exec, TList)
00117         {
00118             typedef typename TList::Head Head;
00119             typedef typename TList::Tail Tail;
00120             
00121             if (Head* p1 = dynamic_cast<Head*>(&lhs))
00122             {
00123                 return DispatchRhs(*p1, rhs, exec, TypesRhs());
00124             }
00125             return DispatchLhs(lhs, rhs, exec, Tail());
00126         }
00127 
00128     public:
00129         static ResultType Go(BaseLhs& lhs, BaseRhs& rhs,
00130             Executor exec)
00131         { return DispatchLhs(lhs, rhs, exec, TypesLhs()); }
00132     };

00173     // Non-inline to reduce compile time overhead...
00174     template <class BaseLhs, class BaseRhs, 
00175                 typename ResultType, typename CallbackType>
00176     void BasicDispatcher<BaseLhs,BaseRhs,ResultType,CallbackType>
00177          ::DoAdd(TypeInfo lhs, TypeInfo rhs, CallbackType fun)
00178     {
00179         callbackMap_[KeyType(lhs, rhs)] = fun;
00180     }
00181         
00182     template <class BaseLhs, class BaseRhs, 
00183                 typename ResultType, typename CallbackType>
00184     bool BasicDispatcher<BaseLhs,BaseRhs,ResultType,CallbackType>
00185          ::DoRemove(TypeInfo lhs, TypeInfo rhs)
00186     {
00187         return callbackMap_.erase(KeyType(lhs, rhs)) == 1;
00188     }

00190     template <class BaseLhs, class BaseRhs, 
00191                 typename ResultType, typename CallbackType>
00192     ResultType BasicDispatcher<BaseLhs,BaseRhs,ResultType,CallbackType>
00193                ::Go(BaseLhs& lhs, BaseRhs& rhs)
00194     {
00195         typename MapType::key_type k(typeid(lhs),typeid(rhs));
00196         typename MapType::iterator i = callbackMap_.find(k);
00197         if (i == callbackMap_.end())
00198         {
00199                 throw std::runtime_error("Function not found");
00200         }
00201         return (i->second)(lhs, rhs);
00202     }
