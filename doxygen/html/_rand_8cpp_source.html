00069 /* initializes mt[N] with a seed */
00070 void init_genrand(unsigned long s)
00071 {
00072     mt[0]= s & 0xffffffffUL;
00073     for (mti=1; mti<N; mti++) {
00074         mt[mti] = 
00075             (1812433253UL * (mt[mti-1] ^ (mt[mti-1] >> 30)) + mti); 
00076         /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
00077         /* In the previous versions, MSBs of the seed affect   */
00078         /* only MSBs of the array mt[].                        */
00079         /* 2002/01/09 modified by Makoto Matsumoto             */
00080         mt[mti] &= 0xffffffffUL;
00081         /* for >32 bit machines */
00082     }
00083 }

00089 void init_by_array(unsigned long init_key[], int key_length)
00090 {
00091     int i, j, k;
00092     init_genrand(19650218UL);

00097           + init_key[j] + j; /* non linear */

00099         i++; j++;
00100         if (i>=N) { mt[0] = mt[N-1]; i=1; }

00114 /* generates a random number on [0, 0xffffffff]-interval */
00115 unsigned long genrand_int32(void)
00116 {
00117     unsigned long y;
00118     static unsigned long mag01[2]={0x0UL, MATRIX_A};
00119     /* mag01[x] = x * MATRIX_A  for x=0, 1 */
00120 
00121     if (mti >= N) { /* generate N words at one time */
00122         int kk;
00123 
00124         if (mti == N+1)   /* if init_genrand() has not been called, */
00125             init_genrand(5489UL); /* a default initial seed is used */
00126 
00127         for (kk=0;kk<N-M;kk++) {
00128             y = (mt[kk]&UPPER_MASK)|(mt[kk+1]&LOWER_MASK);
00129             mt[kk] = mt[kk+M] ^ (y >> 1) ^ mag01[y & 0x1UL];
00130         }
00131         for (;kk<N-1;kk++) {
00132             y = (mt[kk]&UPPER_MASK)|(mt[kk+1]&LOWER_MASK);
00133             mt[kk] = mt[kk+(M-N)] ^ (y >> 1) ^ mag01[y & 0x1UL];
00134         }
00135         y = (mt[N-1]&UPPER_MASK)|(mt[0]&LOWER_MASK);
00136         mt[N-1] = mt[M-1] ^ (y >> 1) ^ mag01[y & 0x1UL];
00137 
00138         mti = 0;
00139     }
00140   
00141     y = mt[mti++];
00142 
00143     /* Tempering */
00144     y ^= (y >> 11);
00145     y ^= (y << 7) & 0x9d2c5680UL;
00146     y ^= (y << 15) & 0xefc60000UL;
00147     y ^= (y >> 18);
00148 
00149     return y;
00150 }
