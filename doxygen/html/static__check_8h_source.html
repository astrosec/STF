00039 // macro STATIC_CHECK
00040 // Invocation: STATIC_CHECK(expr, id)
00041 // where:
00042 // expr is a compile-time integral or pointer expression
00043 // id is a C++ identifier that does not need to be defined
00044 // If expr is zero, id will appear in a compile-time error message.

00047 #define STATIC_CHECK(expr, msg) \
00048     { Loki::CompileTimeError<((expr) != 0)> ERROR_##msg; (void)ERROR_##msg; }
