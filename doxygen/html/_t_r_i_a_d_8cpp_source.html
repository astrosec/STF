00023 datatype::Quaternion estimate_(
00024         datatype::StaticVector<3> v1, datatype::StaticVector<3> v2, 
00025         datatype::StaticVector<3> w1, datatype::StaticVector<3> w2)
00026 {
00027         datatype::StaticVector<3> r1 = v1;
00028         datatype::StaticVector<3> s1 = w1;
00029         datatype::StaticVector<3> r2 = (r1 % v2) / (r1 % v2).norm(2);
00030         datatype::StaticVector<3> s2 = (s1 % w2) / (s1 % w2).norm(2);
00031         datatype::StaticVector<3> r3 = r1 % r2;
00032         datatype::StaticVector<3> s3 = s1 % s2;
00033         datatype::DCM dcm;
00034         for(int i = 0; i < 3; i++){
00035                 for(int j = 0; j < 3; j++){
00036                         dcm[i][j] = s1[i] * r1[j] + s2[i] * r2[j] + s3[i] * r3[j];
00037                         dcm[i][j] = s1[i] * r1[j] + s2[i] * r2[j] + s3[i] * r3[j];
00038                         dcm[i][j] = s1[i] * r1[j] + s2[i] * r2[j] + s3[i] * r3[j];
00039                 }
00040         }
00041         return datatype::TypeConverter::toQuaternion(dcm);
