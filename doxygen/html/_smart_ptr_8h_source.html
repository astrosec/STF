00066         {}

00070         {}

00073         
00074         PointerType operator->() const { return pointee_; }
00075         
00076         ReferenceType operator*() const { return *pointee_; }
00077         
00078         void Swap(DefaultSPStorage& rhs)
00079         { std::swap(pointee_, rhs.pointee_); }
00080     
00081         // Accessors
00082         friend inline PointerType GetImpl(const DefaultSPStorage& sp)
00083         { return sp.pointee_; }

00087

00089         { return sp.pointee_; }
00090

00092         // Destroys the data stored

00095         { delete pointee_; }
00096         
00097         // Default value to initialize the pointer
00098         static StoredType Default()
00099         { return 0; }

00283     namespace Private
00284     {
00285         class RefLinkedBase
00286         {
00287         public:
00288             RefLinkedBase() 
00289             { prev_ = next_ = this; }
00290             
00291             RefLinkedBase(const RefLinkedBase& rhs) 
00292             {
00293                 prev_ = &rhs;
00294                 next_ = rhs.next_;
00295                 prev_->next_ = this;
00296                 next_->prev_ = this;
00297             }
00298             
00299             bool Release()
00300             {
00301                 if (next_ == this)
00302                 {   
00303                     assert(prev_ == this);
00304                     return true;
00305                 }
00306                 prev_->next_ = next_;
00307                 next_->prev_ = prev_;
00308                 return false;
00309             }
00310             
00311             void Swap(RefLinkedBase& rhs)
00312             {
00313                 if (next_ == this)
00314                 {
00315                     assert(prev_ == this);
00316                     if (rhs.next_ == &rhs)
00317                     {
00318                         assert(rhs.prev_ == &rhs);
00319                         // both lists are empty, nothing 2 do
00320                         return;
00321                     }
00322                     prev_ = rhs.prev_;
00323                     next_ = rhs.next_;
00324                     prev_->next_ = next_->prev_ = this;
00325                     rhs.next_ = rhs.prev_ = &rhs;
00326                     return;
00327                 }
00328                 if (rhs.next_ == &rhs)
00329                 {
00330                     rhs.Swap(*this);
00331                     return;
00332                 }
00333                 std::swap(prev_, rhs.prev_);
00334                 std::swap(next_, rhs.next_);
00335                 std::swap(prev_->next_, rhs.prev_->next_);
00336                 std::swap(next_->prev_, rhs.next_->prev_);
00337             }
00338                 
00339             enum { destructiveCopy = false };
00340 
00341         private:
00342             mutable const RefLinkedBase* prev_;
00343             mutable const RefLinkedBase* next_;
00344         };
00345     }
