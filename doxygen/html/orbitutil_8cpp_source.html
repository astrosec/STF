00101 int TransGeoToGAxis(double lon, double lat, double alt, double* pos)
00102 {
00103         double n;
00104         double clat = cos(lat);
00105         double slat = sin(lat);
00106         double clon = cos(lon);
00107         double slon = sin(lon);
00108 
00109         n = XKMPER / pow(1-F*(2-F)*slat*slat, 0.5 );
00110 
00111         pos[0] = (n+alt) * clat * clon;
00112         pos[1] = (n+alt) * clat * slon;
00113         pos[2] = (n+alt-F*(2-F)*n) * slat;
00114 
00115         return 0;
00116 }

00133 void TransECIToGeo(double gmst, double *pos, double *lon, double* lat, double* alt)
00134 {
00135         /* Procedure Calculate_LatLonAlt will calculate the geodetic  */
00136         /* position of an object given its ECI position pos and time. */
00137         /* It is intended to be used to determine the ground track of */
00138         /* a satellite.  The calculations  assume the earth to be an  */
00139         /* oblate spheroid as defined in WGS '72.                     */
00140 
00141         /* Reference:  The 1992 Astronomical Almanac, page K12. */
00142 
00143         double r, e2, phi, c;
00144         double theta;
00145 
00146         theta=AcTan(pos[1], pos[0]); /* radians */
00147         *lon=FMod2p(theta-gmst); /* radians */  //東経が正？
00148         r=sqrt(pos[0]*pos[0]+pos[1]*pos[1]);
00149         e2=F*(2-F);
00150         *lat=AcTan(pos[2], r); /* radians */
00151 
00152         do
00153         {
00154                 phi=*lat;
00155                 c=1/sqrt(1-e2*sin(phi)*sin(phi));
00156                 *lat=AcTan(pos[2]+XKMPER*c*e2*sin(phi), r);
00157 
00158         } while (fabs(*lat-phi)>=1E-10);
00159 
00160         *alt=r/cos(*lat)-XKMPER*c; /* kilometers */     //楕円体高さ
00161 
00162         if (*lat>PIO2)
00163                 *lat-=TWOPI;
00164 }

00174 void CalculateSolarPosition(double jd, double *solar_vector)
00175 {
00176         double l, m, c, theta, epsi, t, r, nu, e;
00177 
00178         t = (jd-2415020)/36525;
00179         l = 279.69668 + 36000.76892 * t + 0.0003025 *t*t;
00180         m = 358.475833 + 35999.04975 * t - 0.00015 *t*t + 0.0000033 *t*t*t;
00181         c = ( (1.919460 - 0.004789 * t - 0.000014 *t*t) * sin(m*DEG2RAD)
00182                 + (0.020094 - 0.0001 *t) * sin(2*m*DEG2RAD) + 0.000293 * sin(3*m*DEG2RAD) );
00183         theta = l + c - 0.00556 - 0.00479*sin((259.18-1934.142*t)*DEG2RAD);
00184         epsi = 23.452294 - 0.0130125*t - 0.00000164*t*t - 0.000000503*t*t*t + 0.00256*cos((259.18-1934.142*t)*DEG2RAD);
00185 
00186         e = 0.01675104 - 0.0000418*t - 0.000000126*t*t;
00187         nu = m+c;
00188         r = 1.00000023 * (1-e*e) / (1+ e*cos(nu*DEG2RAD));
00189         r *= AU;
00190 
00191         solar_vector[0]=r*cos(theta*DEG2RAD);
00192         solar_vector[1]=r*sin(theta*DEG2RAD)*cos(epsi*DEG2RAD);
00193         solar_vector[2]=r*sin(theta*DEG2RAD)*sin(epsi*DEG2RAD);
