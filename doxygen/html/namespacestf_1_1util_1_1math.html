                                                  {
    stf_assert(m.rows() == m.cols());//正方行列のみ計算可能

    datatype::Matrix result(m.rows(), m.cols());
    result.unitize();

    int k = 1;
    datatype::Matrix m_n = m;

    while(k < n){
        //指数関数I+A+A^2/2!+...のk番目の項を計算
        int i = k - 1;
        int f = k;//k!

        while(i > 0){
            m_n *= m;
            f *= i;
            i--;
        }
        
        result += m_n / f;
        m_n = m;
        k++;
    }

    return result;
}
